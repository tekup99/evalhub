# 1. Çevresel değişkenleri (GPU, RAM vb.) .env dosyasından çekiyoruz
source scripts/unified_pipeline.env

# 2. Değişkenleri tanımlıyoruz
TARGET_MODEL="Qwen3.5-4B-Base"
JUDGE_MODEL="google/gemma-4-26B-A4B-it"
BENCHMARK="aime2026"
TEMPERATURE="0.6"
JUDGE_TEMP="0.6"
MAX_COMPLETION_TOKENS="16384"
FILTERED_FILE="data/passatk_filtered/Qwen3.5-4B-Base/aime2026_t0.6_max16384_corrects.jsonl"

# 3. Judge Worker'ı sbatch ile başlatıp Job ID'sini alıyoruz
judge_job=$(sbatch --parsable \
    --job-name="jdg_manual" \
    --output="logs/jdg_manual_%j.out" \
    --error="logs/jdg_manual_%j.err" \
    --ntasks=1 --cpus-per-task="${JUDGE_SLURM_CPUS}" --mem="${JUDGE_SLURM_MEM}" \
    --gres="${JUDGE_SLURM_GRES}" --time="${JUDGE_SLURM_TIME}" \
    --export=ALL,RUN_MODE="judge_worker",JUDGE_MODEL="$JUDGE_MODEL",TARGET_MODEL="$TARGET_MODEL",BENCHMARK="$BENCHMARK",TEMPERATURE="$TEMPERATURE",MAX_COMPLETION_TOKENS="$MAX_COMPLETION_TOKENS",FILTERED_FILE="$FILTERED_FILE",JUDGE_TEMP="$JUDGE_TEMP" \
    scripts/run_unified_eval.sh)

echo "Baslatilan Judge Job ID: $judge_job"

# 4. Post Worker'ı Judge işlemine bağımlı (--dependency=afterok) olarak başlatıyoruz
post_job=$(sbatch --parsable --dependency=afterok:"${judge_job}" \
    --job-name="post_manual" \
    --output="logs/post_manual_%j.out" \
    --ntasks=1 --cpus-per-task="${POST_SLURM_CPUS}" --mem="${POST_SLURM_MEM}" \
    --time="${POST_SLURM_TIME}" \
    --export=ALL,RUN_MODE="post_worker",JUDGE_MODEL="$JUDGE_MODEL",TARGET_MODEL="$TARGET_MODEL",BENCHMARK="$BENCHMARK",TEMPERATURE="$TEMPERATURE",MAX_COMPLETION_TOKENS="$MAX_COMPLETION_TOKENS",JUDGE_TEMP="$JUDGE_TEMP" \
    scripts/run_unified_eval.sh)

echo "Baslatilan Post Job ID: $post_job (Judge bitmesini bekleyecek)"
