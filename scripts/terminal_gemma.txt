# 1. Gerekli ayarları yükle
source scripts/gemma4_pipeline.env
SCRIPT_PATH="./scripts/run_gemma4_eval.sh"

# 2. Değişkenleri tanımla
BENCHMARK="aime2026"
B_TEMP="0.6"
B_MAX=16384
J_MODEL="google/gemma-4-26B-A4B-it"
J_TEMP="0.6"

# 3. Belirttiğin 4 model için döngüyü çalıştır
for MODEL in "google/gemma-4-E2B" "google/gemma-4-E2B-it" "google/gemma-4-E4B-it" "google/gemma-4-E4B"; do
    CLEAN_MODEL=$(basename "$MODEL")
    # Extract worker tarafından önceden oluşturulmuş filtrelenmiş dosya yolu
    FILTERED_FILE="$(pwd)/data/passatk_filtered/${CLEAN_MODEL}/${BENCHMARK}_t${B_TEMP}_max${B_MAX}_corrects.jsonl"
    
    echo "[INFO] $CLEAN_MODEL için Judge işi gönderiliyor..."
    
    # Judge Worker: Sadece judge modelini ayağa kaldırır ve puanlama yapar
    JUDGE_JOB=$(sbatch --parsable \
        --job-name="jdg_${CLEAN_MODEL}" \
        --output="logs/jdg_${CLEAN_MODEL}_%j.out" \
        --error="logs/jdg_${CLEAN_MODEL}_%j.err" \
        --ntasks=1 --cpus-per-task="${JUDGE_SLURM_CPUS}" --mem="${JUDGE_SLURM_MEM}" \
        --gres="${JUDGE_SLURM_GRES}" --time="${JUDGE_SLURM_TIME}" \
        $([[ -n "${JUDGE_SLURM_NODELIST:-}" ]] && echo "--nodelist=$JUDGE_SLURM_NODELIST") \
        --export=ALL,RUN_MODE="judge_worker",JUDGE_MODEL="$J_MODEL",TARGET_MODEL="$MODEL",BENCHMARK="$BENCHMARK",TEMPERATURE="$B_TEMP",MAX_COMPLETION_TOKENS="$B_MAX",FILTERED_FILE="$FILTERED_FILE",JUDGE_TEMP="$J_TEMP",HF_TOKEN="${HF_TOKEN}",PROJECT_ROOT="$(pwd)" \
        "$SCRIPT_PATH")

    echo "[INFO] $CLEAN_MODEL için Post işi gönderiliyor (Bağımlılık: $JUDGE_JOB)..."
    
    # Post Worker: Judge sonuçlarını toplar ve metrikleri hesaplar (Judge bittikten sonra başlar)
    sbatch --dependency=afterok:"${JUDGE_JOB}" \
        --job-name="post_${CLEAN_MODEL}" \
        --output="logs/post_${CLEAN_MODEL}_%j.out" \
        --ntasks=1 --cpus-per-task="${POST_SLURM_CPUS}" --mem="${POST_SLURM_MEM}" \
        --time="${POST_SLURM_TIME}" \
        --export=ALL,RUN_MODE="post_worker",JUDGE_MODEL="$J_MODEL",TARGET_MODEL="$MODEL",BENCHMARK="$BENCHMARK",TEMPERATURE="$B_TEMP",MAX_COMPLETION_TOKENS="$B_MAX",JUDGE_TEMP="$J_TEMP",PROJECT_ROOT="$(pwd)" \
        "$SCRIPT_PATH"
done
